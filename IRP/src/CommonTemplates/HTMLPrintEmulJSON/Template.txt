            height: 1px;
            background-color: grey;
            margin: 10px 0;
        }
        .colon-line {
            display: flex;
            justify-content: space-between;
        }
        .dotted-line {
            flex-grow: 1;
            height: 1px;

            background-size: 10px 1px;
            align-self: flex-end;

<body>
    <div class="receipts" id="receipts"></div>
    <script>
        function clearAll() {
            document.getElementById('receipts').innerHTML = '';
        }

        function fillData(jsonData) {
            let data = JSON.parse(jsonData);
            let receiptDiv = document.createElement('div');
            receiptDiv.className = 'receipt';
            data.forEach(line => {
                let div = document.createElement('div');
                if (line.includes(':')) {
                    div.className = 'colon-line';
                    let parts = line.split(':');
                    let leftSpan = document.createElement('span');
                    let rightSpan = document.createElement('span');
                    let dottedLine = document.createElement('div');
                    dottedLine.className = 'dotted-line';
                    leftSpan.innerText = parts[0] + ':';
                    rightSpan.innerText = parts[1];
                    div.appendChild(leftSpan);
                    div.appendChild(dottedLine);
                    div.appendChild(rightSpan);
                } else {
                    div.innerText = line;
                }
                receiptDiv.appendChild(div);
            });
            document.getElementById('receipts').appendChild(receiptDiv);
            let separator = document.createElement('div');
            separator.className = 'separator';
            document.getElementById('receipts').appendChild(separator);
        }
    </script>
